            SELECT fagency.field_p_agency_target_id AS agency_id
            
                , SUM(1) AS total_cnt

                , SUM((CASE WHEN (d.status = 1) THEN 1 ELSE 0 END)) AS old_published_yes_cnt
                , SUM((CASE WHEN (d.status = 0) THEN 1 ELSE 0 END)) AS old_published_not_cnt
                , SUM((CASE WHEN (d.status IS NULL) THEN 1 WHEN (d.status NOT IN (1,0)) THEN 1 ELSE 0 END)) AS old_published_unk_cnt

                , SUM((CASE WHEN ((aid.nid IS NOT NULL) AND (aid.status IS NULL)) THEN 1 ELSE 0 END)) AS active_yes_cnt
                , SUM((CASE WHEN ((aid.nid IS NOT NULL) AND (aid.status IS NOT NULL)) THEN 1 ELSE 0 END)) AS active_not_cnt
                , SUM((CASE WHEN (aid.nid IS NULL) THEN 1 ELSE 0 END)) AS active_unk_cnt

                , SUM((CASE WHEN (fpubl.field_p_publ_value = 1) THEN 1 ELSE 0 END)) AS publ_yes_cnt
                , SUM((CASE WHEN (fpubl.field_p_publ_value = 0) THEN 1 ELSE 0 END)) AS publ_not_cnt
                , SUM((CASE WHEN (fpubl.field_p_publ_value IS NULL) THEN 1 WHEN (fpubl.field_p_publ_value NOT IN (1,0)) THEN 1 ELSE 0 END)) AS publ_unk_cnt

                , SUM((CASE WHEN (fcluster.field_p_cluster_value = 1) THEN 1 ELSE 0 END)) AS cluster_yes_cnt
                , SUM((CASE WHEN (fcluster.field_p_cluster_value = 0) THEN 1 ELSE 0 END)) AS cluster_not_cnt
                , SUM((CASE WHEN (fcluster.field_p_cluster_value IS NULL) THEN 1 WHEN (fcluster.field_p_cluster_value NOT IN (1,0)) THEN 1 ELSE 0 END)) AS cluster_unk_cnt

                , SUM((CASE WHEN (fcheckin.field_p_checkin_value = 1) THEN 1 ELSE 0 END)) AS checkin_yes_cnt
                , SUM((CASE WHEN (fcheckin.field_p_checkin_value = 0) THEN 1 ELSE 0 END)) AS checkin_not_cnt
                , SUM((CASE WHEN (fcheckin.field_p_checkin_value IS NULL) THEN 1 WHEN (fcheckin.field_p_checkin_value NOT IN (1,0)) THEN 1 ELSE 0 END)) AS checkin_unk_cnt

                , SUM((CASE WHEN (fflagman.field_p_flagman_value = 1) THEN 1 ELSE 0 END)) AS old_flagman_yes_cnt
                , SUM((CASE WHEN (fflagman.field_p_flagman_value = 0) THEN 1 ELSE 0 END)) AS old_flagman_not_cnt
                , SUM((CASE WHEN (fflagman.field_p_flagman_value IS NULL) THEN 1 WHEN (fflagman.field_p_flagman_value NOT IN (1,0)) THEN 1 ELSE 0 END)) AS old_flagman_unk_cnt

                , SUM((CASE WHEN (fexclusive.field_p_exclusive_value = 1) THEN 1 ELSE 0 END)) AS exclusive_yes_cnt
                , SUM((CASE WHEN (fexclusive.field_p_exclusive_value = 0) THEN 1 ELSE 0 END)) AS exclusive_not_cnt
                , SUM((CASE WHEN (fexclusive.field_p_exclusive_value IS NULL) THEN 1 WHEN (fexclusive.field_p_exclusive_value NOT IN (1,0)) THEN 1 ELSE 0 END)) AS exclusive_unk_cnt



                , SUM((CASE WHEN (fcomplex.field_p_complex_target_id IS NOT NULL) THEN 1 ELSE 0 END)) AS complex_yes_cnt
                , SUM((CASE WHEN (fcomplex.field_p_complex_target_id IS NULL) THEN 1 ELSE 0 END)) AS complex_not_cnt
                , 0 AS complex_unk_cnt

                , SUM((CASE WHEN (fplan.field_p_plan_target_id IS NOT NULL) THEN 1 ELSE 0 END)) AS plan_yes_cnt
                , SUM((CASE WHEN (fplan.field_p_plan_target_id IS NULL) THEN 1 ELSE 0 END)) AS plan_not_cnt
                , 0 AS plan_unk_cnt

                , SUM((CASE WHEN (fplanhh.field_p_plan_hh_target_id IS NOT NULL) THEN 1 ELSE 0 END)) AS planhh_yes_cnt
                , SUM((CASE WHEN (fplanhh.field_p_plan_hh_target_id IS NULL) THEN 1 ELSE 0 END)) AS planhh_not_cnt
                , 0 AS planhh_unk_cnt



                , SUM((CASE WHEN (frating.field_p_rating_value IS NOT NULL) THEN 1 ELSE 0 END)) AS rating_yes_cnt
                , SUM((CASE WHEN (frating.field_p_rating_value IS NULL) THEN 1 ELSE 0 END)) AS rating_not_cnt
                , 0 AS rating_unk_cnt
                , COALESCE(SUM(frating.field_p_rating_value),0) AS rating_yes_sum
                , ROUND(COALESCE(AVG(frating.field_p_rating_value),0),2) AS rating_yes_avg

                , SUM((CASE 
                    WHEN (true
                        AND (fmain.field_p_main_property_target_id IS NULL) -- он ни на кого не ссылается 
                        AND (fchild_node.field_p_main_property_target_id IS NULL) -- на него никто не ссылается
                    ) THEN 1 
                    ELSE 0 
                END)) AS unique_yes_cnt
                , 0 AS unique_not_cnt
                , SUM((CASE     
                    WHEN (true
                        AND (fmain.field_p_main_property_target_id IS NOT NULL) -- он на кого-то ссылается
                        AND (fchild_node.field_p_main_property_target_id IS NOT NULL) -- на него кто-то ссылается
                    ) THEN 1 
                    ELSE 0 
                END)) AS unique_unk_cnt
                , GROUP_CONCAT((CASE
                    WHEN (true
                        AND (fmain.field_p_main_property_target_id IS NOT NULL) -- он на кого-то ссылается
                        AND (fchild_node.field_p_main_property_target_id IS NOT NULL) -- на него кто-то ссылается
                    ) THEN n.nid
                    ELSE NULL
                END)) AS unique_unk_nids
                

                , SUM((CASE 
                    WHEN (true
                        AND (fflagman.field_p_flagman_value = 1) -- у него стоит галка флагман
                        AND (fmain.field_p_main_property_target_id IS NULL) -- он ни на кого не ссылается 
                        AND (fchild_node.field_p_main_property_target_id IS NOT NULL) -- на него кто-то ссылается
                    ) THEN 1 
                    ELSE 0 
                END)) AS flagman_yes_cnt
                , 0 AS flagman_not_cnt
                , SUM((CASE     
                    WHEN (fflagman.field_p_flagman_value IS NULL) THEN 1
                    WHEN (fflagman.field_p_flagman_value NOT IN (0,1)) THEN 1
                    WHEN (true
                        AND (fmain.field_p_main_property_target_id IS NOT NULL) -- он на кого-то ссылается
                        AND (fchild_node.field_p_main_property_target_id IS NOT NULL) -- на него кто-то ссылается
                    ) THEN 1 
                    ELSE 0 
                END)) AS flagman_unk_cnt
                
            FROM node AS n
                INNER JOIN node_field_data AS d ON (true
                    AND (d.nid = n.nid)
                    AND (d.langcode = n.langcode)
                    AND (d.type = n.type)
                    AND (d.vid = n.vid)
                )
                INNER JOIN node__field_p_type AS ftype ON (true
                    AND (ftype.entity_id = n.nid)
                    AND (ftype.deleted = 0)
                    AND (ftype.delta = 0)
                    AND (ftype.langcode = n.langcode)
                )
                LEFT JOIN node__field_p_clusterlot AS fclusterlot ON (true
                    AND (fclusterlot.entity_id = n.nid)
                    AND (fclusterlot.deleted = 0)
                    AND (fclusterlot.delta = 0)
                    AND (fclusterlot.langcode = n.langcode)
                )
                LEFT JOIN node__field_p_agency AS fagency ON (true
                    AND (fagency.entity_id = n.nid)
                    AND (fagency.deleted = 0)
                    AND (fagency.delta = 0)
                    AND (fagency.langcode = n.langcode)
                )
                LEFT JOIN node__field_p_typeinsert AS ftypeinsert ON (true
                    AND (ftypeinsert.entity_id = n.nid)
                    AND (ftypeinsert.deleted = 0)
                    AND (ftypeinsert.delta = 0)
                    AND (ftypeinsert.langcode = n.langcode)
                )
                
                LEFT JOIN async_import_data AS aid ON (aid.nid = n.nid)
                
                LEFT JOIN node__field_p_publ AS fpubl ON (true
                    AND (fpubl.entity_id = n.nid)
                    AND (fpubl.deleted = 0)
                    AND (fpubl.delta = 0)
                    AND (fpubl.langcode = n.langcode)
                )


                LEFT JOIN node__field_p_cluster AS fcluster ON (true
                    AND (fcluster.entity_id = n.nid)
                    AND (fcluster.deleted = 0)
                    AND (fcluster.delta = 0)
                    AND (fcluster.langcode = n.langcode)
                )
                LEFT JOIN node__field_p_checkin AS fcheckin ON (true
                    AND (fcheckin.entity_id = n.nid)
                    AND (fcheckin.deleted = 0)
                    AND (fcheckin.delta = 0)
                    AND (fcheckin.langcode = n.langcode)
                )
                LEFT JOIN node__field_p_flagman AS fflagman ON (true
                    AND (fflagman.entity_id = n.nid)
                    AND (fflagman.deleted = 0)
                    AND (fflagman.delta = 0)
                    AND (fflagman.langcode = n.langcode)
                )
                LEFT JOIN node__field_p_exclusive AS fexclusive ON (true
                    AND (fexclusive.entity_id = n.nid)
                    AND (fexclusive.deleted = 0)
                    AND (fexclusive.delta = 0)
                    AND (fexclusive.langcode = n.langcode)
                )




                LEFT JOIN node__field_p_complex AS fcomplex ON (true
                    AND (fcomplex.entity_id = n.nid)
                    AND (fcomplex.deleted = 0)
                    AND (fcomplex.delta = 0)
                    AND (fcomplex.langcode = n.langcode)
                )
                LEFT JOIN node__field_p_plan AS fplan ON (true
                    AND (fplan.entity_id = n.nid)
                    AND (fplan.deleted = 0)
                    AND (fplan.delta = 0)
                    AND (fplan.langcode = n.langcode)
                )
                LEFT JOIN node__field_p_plan_hh AS fplanhh ON (true
                    AND (fplanhh.entity_id = n.nid)
                    AND (fplanhh.deleted = 0)
                    AND (fplanhh.delta = 0)
                    AND (fplanhh.langcode = n.langcode)
                )
                
                
                
                LEFT JOIN node__field_p_rating AS frating ON (true
                    AND (frating.entity_id = n.nid)
                    AND (frating.deleted = 0)
                    AND (frating.delta = 0)
                    AND (frating.langcode = n.langcode)
                )
                
                
                LEFT JOIN node__field_p_main_property AS fmain ON (true
                    AND (fmain.entity_id = n.nid)
                    AND (fmain.deleted = 0)
                    AND (fmain.delta = 0)
                    AND (fmain.langcode = n.langcode)
                )
                LEFT JOIN (
                    SELECT field_p_main_property_target_id, langcode, COUNT(*) AS cnt
                    FROM node__field_p_main_property
                    WHERE (true
                        AND (bundle = 'property')
                        AND (deleted = 0)
                    )
                    GROUP BY field_p_main_property_target_id, langcode
                ) AS fchild_node ON (true
                    AND (fchild_node.field_p_main_property_target_id = n.nid)
                    AND (fchild_node.langcode = n.langcode)
                )
                
            WHERE (true 
                AND (n.langcode = 'ru')
                AND (n.type = 'property')
                #AND (d.status = 1) # опубликованные
                AND (ftype.field_p_type_target_id = 1) # квартиры
                AND ((fclusterlot.field_p_clusterlot_value IS NULL) OR (fclusterlot.field_p_clusterlot_value = 0))
                
                
            )
            GROUP BY fagency.field_p_agency_target_id
            ORDER BY active_yes_cnt DESC, agency_id ASC
